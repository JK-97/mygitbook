3.1.装饰器

3.1.装饰器

函数装饰器

import time


def log_time(func):
    def _log(*args, **kwargs):
        beg = time.time()
        res = func(*args, **kwargs)
        print('use time: {}'.format(time.time() - beg))
        return res

    return _log


@log_time
def mysleep():
    time.sleep(1)


mysleep()

类装饰器

import time

class LogTime:
     def __call__(self,func):
        def _log(*args, **kwargs):
            beg = time.time()
            res = func(*args, **kwargs)
            print('use time: {}'.format(time.time() - beg))
            return res

        return _log


@LogTime()
def mysleep():
    time.sleep(1)


mysleep()
